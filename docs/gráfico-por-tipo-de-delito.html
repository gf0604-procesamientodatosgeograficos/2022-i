Chapter 17 Gráfico por tipo de delito

grafico_delitos_x_delito <-
  estadisticas_policiales %>%
  count(Delito) %>%
  ggplot(aes(x = reorder(Delito, n), y = n)) +
  geom_bar(stat = "identity") +
  coord_flip() +
  theme_minimal()

grafico_delitos_x_delito %>%
  ggplotly() %>% 
  config(locale = 'es') # traducción de etiquetas al español
